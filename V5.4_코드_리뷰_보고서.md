### **[심각도: 상] S-01 — CORS `origin: true` 설정**

**파일:** `backend/src/app.js`, lines 59–65, 78–81
**카테고리:** 보안 / CORS 설정 오류

**문제:**
```js
// app.js — 현재 코드
const io = new Server(server, {
  cors: { origin: true, credentials: true }  // ← 모든 출처 반사
})
app.use(cors({ origin: true, credentials: true }))
```
`origin: true`는 `*`보다 위험합니다. 요청 `Origin` 헤더를 그대로 반사하므로 `credentials: true`와 결합하면 어떤 악성 웹페이지에서도 인증 쿠키를 포함한 크로스 오리진 요청이 가능해집니다.

**수정 코드:**
```js
const ALLOWED_ORIGINS = (process.env.ALLOWED_ORIGINS || 'http://localhost:5173')
  .split(',')
  .map(o => o.trim())

app.use(cors({
  origin: (origin, callback) => {
    if (!origin || ALLOWED_ORIGINS.includes(origin)) {
      callback(null, true)
    } else {
      callback(new Error(`CORS policy: origin ${origin} not allowed`))
    }
  },
  credentials: true,
}))
```

### **[심각도: 상] S-05 — 파일 업로드 MIME 검증 없음 (magic bytes 미확인)**

**파일:** `backend/src/middleware/upload.js`, lines 26–38
**카테고리:** 보안 / 파일 업로드

**문제:**
```js
// upload.js — 현재 코드
fileFilter: (req, file, cb) => {
  const allowedMimes = ['image/jpeg', 'image/png', ..., 'text/html', 'application/zip']
  if (allowedMimes.includes(file.mimetype)) {
    cb(null, true)  // ← 클라이언트가 보낸 mimetype만 신뢰
  }
}
```
클라이언트가 `Content-Type: image/jpeg`로 위장한 PHP 셸이나 HTML XSS 파일을 업로드할 수 있습니다. `text/html`은 같은 도메인에서 XSS를 허용합니다.

**수정 코드:**
```js
const { fileTypeFromBuffer } = require('file-type')

// multer는 메모리에 임시 저장 후 검증
const storage = multer.memoryStorage()

// 저장 후 magic bytes 검증 미들웨어
const validateFileType = async (req, res, next) => {
  if (!req.file) return next()

  const fileType = await fileTypeFromBuffer(req.file.buffer)
  const SAFE_TYPES = ['image/jpeg', 'image/png', 'image/gif', 'image/webp',
                      'video/mp4', 'video/webm', 'audio/mpeg', 'audio/wav',
                      'application/pdf', 'font/woff', 'font/woff2']

  if (!fileType || !SAFE_TYPES.includes(fileType.mime)) {
    return res.status(400).json({ error: `허용되지 않는 파일 형식: ${fileType?.mime || '알 수 없음'}` })
  }

  // 검증된 mimeType으로 덮어씀
  req.file.mimetype = fileType.mime
  next()
}
```

export function formatBytes(bytes: number, decimals = 1): string {
  if (bytes === 0) return '0 B'
  const k = 1024
  const sizes = ['B', 'KB', 'MB', 'GB', 'TB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  return `${parseFloat((bytes / Math.pow(k, i)).toFixed(decimals))} ${sizes[i]}`
}

// components/ErrorBoundary.tsx
import { Component, ErrorInfo, ReactNode } from 'react'

interface Props { children: ReactNode; fallback?: ReactNode }
interface State { hasError: boolean; error?: Error }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    console.error('[ErrorBoundary] Caught:', error, info)
    // Sentry.captureException(error, { extra: info })
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? (
        <div className="flex flex-col items-center justify-center h-screen gap-4">
          <h2 className="text-xl font-bold text-red-600">페이지 오류 발생</h2>
          <p className="text-gray-500">{this.state.error?.message}</p>
          <button onClick={() => this.setState({ hasError: false })} className="btn-primary">
            다시 시도
          </button>
        </div>
      )
    }
    return this.props.children
  }
}

// utils/safeDate.ts
import { format, isValid } from 'date-fns'
import { ko } from 'date-fns/locale'

export function safeFormat(
  dateStr: string | null | undefined,
  fmt: string,
  fallback = '—'
): string {
  if (!dateStr) return fallback
  const date = new Date(dateStr)
  if (!isValid(date)) return fallback
  try {
    return format(date, fmt, { locale: ko })
  } catch {
    return fallback
  }
}
